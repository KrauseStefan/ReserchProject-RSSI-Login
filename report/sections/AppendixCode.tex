\shellcmd{‘sudo hciconfig hci0 leadv’}

\shellcmd{PATH=\$PATH:/home/pi/node-v0.10.26-linux-arm-pi/bin}
\shellcmd{export PATH}
\shellcmd{node <start BLE app>.js}
\shellcmd{sudo node index.js | tee log.csv}
